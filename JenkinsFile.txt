	pipeline
	
	{
		agent any
		
		stages {
			stage ('Build') 
			{
				steps 
				{
					echo "Build creation is completed..."
                    sh "mvn clean"
				}
			}
			
			stage ('Test') 
			{
				steps 
				{
					echo "Testing the project..."
                    sh "mvn test"             
				}
			}
			
            stage ('Compile') 
			{
				steps 
				{
					echo "Compiling a project..."
                    sh "mvn compile"             
				}
			}
			
			stage ('Deploy') 
			{
				steps 
				{
					echo "Deployed to production..."
				}
			}

		}

		}
